            correctLevel: QRCode.CorrectLevel.L
        });
        let details = parseQRCodeContent(qrCodeContent);
        if (details.size > 0) {
            let title = '';
            details.forEach((value, tag) => {
                title += `Tag ${tag} :  ${value.join(', ')}\n`;
            });
            var signedQrCodeDiv = document.getElementById('signedQrCode');
            if (signedQrCodeDiv) {
                signedQrCodeDiv.title = title.trim();
            }
        }
    };
    var qrcodeDiv = document.getElementById('qrcode');
    if (qrcodeDiv) {
        qrcodeDiv.remove();
    };
    qrCodeFooterDiv = document.getElementById('qrCodeFooter');
    if (qrCodeFooterDiv){
        qrCodeFooterDiv.remove();
    };
});
function findParentTable(element) {
    while (element) {
        if (element.tagName === 'TABLE') {
            return element;
        }
        element = element.parentElement;
    }
    return null;
}
function parseQRCodeContent(qrCodeBase64) {
    let details = new Map();
    try {
        let data = atob(qrCodeBase64.replace(/&#x2B;/g, '+'));
        let index = 0;
        while (index < data.length) {
            let tag = data.charCodeAt(index++);
            let length = data.charCodeAt(index++);
            let value = data.substr(index, length);
            index += length;
            let decodedValue;
            if (tag === 8 || tag === 9) {
                decodedValue = [...value].map(c => c.charCodeAt(0).toString(16).padStart(2, '0')).join('');
            } else {
                let decoder = new TextDecoder('utf-8');
                decodedValue = decoder.decode(new Uint8Array([...value].map(c => c.charCodeAt(0))));
            }
            details.set(tag, [decodedValue]);
        }
    } catch (ex) {
        console.error("Error decoding QR code: " + ex.message);
    }
    return details;
}
